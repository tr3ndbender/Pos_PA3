Client-Server-Verbindung + zufälliges Wort via XML

Server/Program.cs

using Server;
using System.Net;
using System.Net.Sockets;
using System.Xml.Linq;
using Server.Models;

// DB beim Start sicherstellen
using var db = new AppDbContext();
db.Database.EnsureCreated();

var listener = new TcpListener(IPAddress.Any, PORT);
listener.Start();
Console.WriteLine("Server läuft auf Port PORT...");

while (true)
{
    var client = listener.AcceptTcpClient();
    _ = Task.Run(() => HandleClient(client));
}

void HandleClient(TcpClient client)
{
    var stream = client.GetStream();
    var reader = new StreamReader(stream);
    var writer = new StreamWriter(stream) { AutoFlush = true };

    // Zufälliges Wort aus DB (nur korrekte Länge)
    using var db = new AppDbContext();
    var allWords = db.Words.ToList().Where(w => w.Text.Length == WORTLAENGE).ToList();
    var randomWord = allWords[new Random().Next(allWords.Count)].Text.ToUpper();

    // Wort als XML senden
    writer.WriteLine($"<Word>{randomWord}</Word>");

    // Auf Guesses warten
    string line;
    while ((line = reader.ReadLine()) != null)
    {
        var guess = XDocument.Parse(line).Root.Value;
        var result = CheckGuess(guess, randomWord);
        writer.WriteLine(result);
    }
}

string CheckGuess(string guess, string word)
{
    var results = new char[word.Length];
    for (int i = 0; i < word.Length; i++)
    {
        if (guess[i] == word[i])             results[i] = 'G'; // Grün
        else if (word.Contains(guess[i]))  results[i] = 'Y'; // Gelb
        else                                  results[i] = 'X'; // Grau
    }
    return $"<Result>{string.Join(",", results)}</Result>";
}

MainWindow.xaml.cs

private void RatenButton_Click(object sender, RoutedEventArgs e)
{
    var guess = GuessInput.Text.ToUpper().Trim();

    // Validierung – muss genau WORTLAENGE Zeichen haben
    if (guess.Length != WORTLAENGE)
    {
        MessageBox.Show("Das Wort muss genau WORTLAENGE Zeichen haben!", "Fehler");
        return;
    }

    // Guess zum Server schicken
    _writer.WriteLine($"<Guess>{guess}</Guess>");

    // Ergebnis empfangen
    var resultXml = _reader.ReadLine();           // "<Result>G,X,Y,...</Result>"
    var colors = XDocument.Parse(resultXml).Root.Value.Split(',');

    // Grid einfärben
    ColorRow(_currentRow, guess, colors);
    _currentRow++;

    // Gewonnen oder verloren?
    if (colors.All(c => c == "G"))
        EndGame(true);
    else if (_currentRow >= ANZAHL_VERSUCHE)
        EndGame(false);

    GuessInput.Clear();
}

MainWindow.xaml.cs – Zellen initialisieren + ColorRow

// Felder oben in der Klasse:
ObservableCollection<Cell> _cells = new();

// Im Konstruktor (nach InitializeComponent):
for (int i = 0; i < ANZAHL_VERSUCHE * WORTLAENGE; i++) _cells.Add(new Cell());
GuessGrid.ItemsSource = _cells;
// ColorRow-Methode:
void ColorRow(int row, string guess, string[] colors)
{
    for (int i = 0; i < WORTLAENGE; i++)
    {
        _cells[row * WORTLAENGE + i].Letter = guess[i].ToString();
        _cells[row * WORTLAENGE + i].Color = colors[i] switch
        {
            "G" => Brushes.Green,
            "Y" => Brushes.Yellow,
            _   => Brushes.Gray
        };
    }
}

MainWindow.xaml.cs – Statistik-Array + EndGame-Methode

// Feld oben in der Klasse:
int[] _stats = new int[ANZAHL_VERSUCHE + 1];
// [0]..[ANZAHL_VERSUCHE-1] = beim 1./2./.../letzten Versuch erraten
// [ANZAHL_VERSUCHE]        = nie erraten
void EndGame(bool won)
{
    // Statistik aktualisieren
    if (won) _stats[_currentRow - 1]++;
    else     _stats[ANZAHL_VERSUCHE]++;

    // Eingabe sperren
    GuessInput.IsEnabled  = false;
    RatenButton.IsEnabled = false;

    // Fenster verbreitern (rechte Spalte aufmachen)
    this.Width *= 2;
    var grid = (Grid)this.Content;
    grid.ColumnDefinitions[1].Width = new GridLength(1, GridUnitType.Star);

    // Statistik einblenden
    StatsPanel.Visibility = Visibility.Visible;
    WordLabel.Content  = $"Das Wort war: {_serverWord}";
    Stat1.Content      = $"Beim 1. Versuch: {_stats[0]}";
    Stat2.Content      = $"Beim 2. Versuch: {_stats[1]}";
    Stat3.Content      = $"Beim 3. Versuch: {_stats[2]}";
    Stat4.Content      = $"Beim 4. Versuch: {_stats[3]}";
    Stat5.Content      = $"Beim 5. Versuch: {_stats[4]}";
    Stat6.Content      = $"Beim 6. Versuch: {_stats[5]}";
    StatNever.Content  = $"Nie erraten: {_stats[ANZAHL_VERSUCHE]}";
}
